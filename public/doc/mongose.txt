const mongooseLib = require('mongoose');
const validatorLib = require('validator');

mongooseLib.connect('mongodb://127.0.0.1:27017/task-manager-api');

const User = mongooseLib.model('User', { 
    name: {
        type: String,
        required: true,
        trim: true
    },
    email: {
        type: String,
        required: true,
        trim: true,
        lowercase: true,
        validate(value) {
            if(!validatorLib.isEmail(value)) {
                throw new Error('Email is invalid');
            }
        }
    },
    password: {
        type: String,
        required: true,
        minlength: 6,
        trim: true,
        validate(value) {
            if(value.toLowerCase().includes('password')) {
                throw new Error('Password should not contain "password" keyword');
            }
        }
    },
    age: {
        type: Number,
        default: 0,
        validate (value) {
            if(value < 0) {
                throw new Error('Age must be a positive number');
            }
        }
    }
});
